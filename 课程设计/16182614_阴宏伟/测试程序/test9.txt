int symbol[7];
int array[7];
int n;
void fullarray(int index){
	int i;
	if(n==index){
		i = 0;
		while(i<n){
			printf(array[i]);
			i = i+1;
		}
		return ;
	}
	i = 0;
	while(i<n){
		if(symbol[i]==0){
			array[index] = i+1;
			symbol[i]=1;
			fullarray(index+1);
			symbol[i]=0;
		}
		i = i+1;
	}
	return (0);
}
